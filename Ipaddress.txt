#include <ctype.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
  char input[16];
  fgets(input, 16, stdin);
  char *str = input;
  int i = 0;
  int dotCount = 0;
  int num = 0;
  int invalid = 0;
  while (str[i] != '\0') {
    while (isdigit(str[i])) {
      if (str[i] == '0' && str[i+1] != '\0' && str[i + 1] != '.') {
        invalid = 1;
        break;
      }
      num = num * 10 + (str[i] - '0');
      str++;
    }    
    if (!(num >= 0 && num <= 255)) {
      invalid = 1;
      break;
    }
    

    if (str[i] == '.') {
      if (str[i + 1] == '.') {
        invalid = 1;
      } else {
        num = 0;
        dotCount++;
      }
    }

    if (dotCount >= 4 ) {
      invalid = 1;
    }
    i++;
  }
  if(dotCount < 3){
    invalid = 1;
  }
  if (invalid) {
    printf("Invalid");
  } else {
    printf("Valid");
  }
  return 0;
}
